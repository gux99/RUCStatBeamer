	\section{块环境}
	\begin{frame}
		\frametitle{插入一般的块}
		\begin{block}{贝叶斯规则}
			我们经常会需要在已知$P(y|x)$时计算$P(x|y)$。幸运的是，如果还知道$P(x)$，
			我们可以用\textbf{贝叶斯规则}（Bayes' rule）来实现这一目的：
			$$
			P(x|y) = \frac{P(y|x) P(x)}{P(y)}
			$$
		\end{block}
	\end{frame}
